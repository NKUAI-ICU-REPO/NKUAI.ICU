#include <iostream>
#include <cstring>
 
using namespace std;
 
int main() {
  int n;
  cin >> n;
  char names[1001][101];
  char tmp[101];
 
  for (int i = 0; i < n; ++i) {
    cin >> names[i];
  }
 
  for (int i = 0; i < n - 1; ++i) {
    for (int j = n - 1; j > i; --j) {
      if (strcmp(names[j], names[j - 1]) < 0) {
        strcpy(tmp, names[j]);
        strcpy(names[j], names[j - 1]);
        strcpy(names[j - 1], tmp);
      } 
    }
  }
 
  for (int i = 0; i < n; ++i) {
    cout << names[i] << endl;
  }
 
  return 0;
}
