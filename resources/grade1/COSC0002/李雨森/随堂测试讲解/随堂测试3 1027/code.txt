#include<iostream>
using namespace std;
int a[11][11];
void yanghui()
{
    for (int i = 0; i <= 10; i++)
    {
        a[i][0] = 1;
        a[i][1] = i;
        a[i][i] = 1;
    }
    for(int i=2;i<=10;i++)
        for (int j = 2; j < i; j++)
        {
            a[i][j] = a[i - 1][j - 1] + a[i-1][j ];
        }
}
 
int main()
{
    int n;
    cin >> n;
    yanghui();
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < i; j++)
        {
            cout << a[i][j] << " ";
        }
        cout<<a[i][i]; 
        cout << endl;
    }
    return 0;
}
