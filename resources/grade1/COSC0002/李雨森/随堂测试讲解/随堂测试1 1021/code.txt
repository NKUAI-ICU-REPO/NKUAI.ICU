#include <iostream>
#include <iomanip>
using namespace std;
int cal[7][5];
bool isleap(int y){
    return (y % 400 == 0 || (y % 4 == 0 && y % 100 != 0));
}
int main(){
    int month[13] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    int y, w, m;
    cin >> y >> w >> m;
    if(isleap(y)) month[2]++;
    if(!(0 <= w && w <= 6 && 1 <= m && m <= 12)){
        cout << "ERROR";
    }else{
        int days = 0;
        for(int i = 1; i < m; i++){
            days += month[i];
        }
        int startw = (days + w) % 7;
        for(int i = 0; i < startw; i++){
            cout << setw(4) << ' ';
        }
        for(int i = 1; i <= month[m]; i++){
            cout << setw(4) << i;
            startw++;
            if(startw == 7 && i != month[m]){
                startw = 0;
                cout << endl;
            }
        }
    }
} // namespace std;
