{

            for(int j=0;s[j]!='\0';j++){
                if(S[m++]!=s[j]){
                    panduan=0;
                    break;
                }
            }

            if(panduan==1){
                    for(int j=0;s[j]!='\0';j++){
                        S[m]='1';

                    }
            }
        }
    }

}
